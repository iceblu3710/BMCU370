136void AS5600_soft_IIC::updata_stu()
137{

139 for (int i = 0; i < numbers; i++)
140 {
141 if (error[i])
142 online[i] = false;
143 else

145 if (!(data[i] & 0x20))
146 {
147 magnet_stu[i] = offline;
148 }
149 else
150 {
151 if (data[i] & 0x10)
152 magnet_stu[i] = low;
153 else if (data[i] & 0x08)
154 magnet_stu[i] = high;
