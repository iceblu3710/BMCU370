133 void Route(UnifiedCommandType type, uint8_t* buffer, uint16_t length) {
134 // Generic Connectivity Update
135 if (type != UnifiedCommandType::None && type != UnifiedCommandType::Error) {
136 ControlLogic::UpdateConnectivity(true);
137 } else if (type == UnifiedCommandType::Error) {
138 ControlLogic::UpdateConnectivity(false);
139 return;
140 }
141
142 // Table Lookup
143 bool found = false;
144 for(int i=0; i<routeTableSize; i++) {
145 if (routeTable[i].type == type) {
146 routeTable[i].handler(buffer, length);
147 found = true;
148 break;
149 }
150 }
151
152 if (!found) {
153 // Handle special Long Packet types that map to same logic?
154 // UnifiedCommandType for Long Packets: MCOnline, ReadFilamentInfo, etc.
155 // These all need parsing via ParseLongPacket.
156 // We can add them to table pointing to HandleLongPacket wrapper.
157 switch(type) {
158 case UnifiedCommandType::MCOnline:
159 case UnifiedCommandType::ReadFilamentInfo:
160 case UnifiedCommandType::SetFilamentInfoType2:
161 case UnifiedCommandType::Version:
162 case UnifiedCommandType::SerialNumber:
163 case UnifiedCommandType::FilamentMotionLong: // Also a long packet
164 HandleLongPacket(buffer, length);
165 break;
166 default:
167 break;
168 }
169 }
170 }

180 void Run() {

182 #ifdef STANDARD_SERIAL
183 // Polling if needed?
184 #else
185 KlipperCLI::Run();
186 #endif

188
189 if (process_ready) {
190 process_ready = false;
191
192 if (ControlLogic::GetBootMode() == ControlLogic::BootMode::Klipper) {
193 #ifdef STANDARD_SERIAL
194 SerialCLI::Parse((char*)process_buffer);
195 #else
196 KlipperCLI::Run(); // KlipperCLI processes its own buffer usually, but verify?
197 // Actually KlipperCLI::Run handles feeding from buffer if using FeedByte?
198 // The callback calls FeedByte.
199 // process_ready logic is for PACKET extraction.
200 // For KlipperCLI, process_ready is not used if we feed directly in callback?
201 // The Init callback calls FeedByte directly.
202 // So process_ready is NOT set for KlipperCLI.
203 // But for SerialCLI it IS set.
204 #endif
205 } else {
206 // Bus Mode
207 Hardware::DelayMS(1);
208 BambuBus_package_type bb_type = BambuBusProtocol::IdentifyPacket(process_buffer, process_len);
209 UnifiedCommandType unified_type = BambuBusProtocol::GetUnifiedType(bb_type);
210
211 if (unified_type != UnifiedCommandType::None) {
212 last_packet_time = Hardware::GetTime();
213 Route(unified_type, process_buffer, process_len);
214 } else if (bb_type == BambuBus_package_type::ERROR) {
215 Route(UnifiedCommandType::Error, process_buffer, process_len);
216 }
217 }
218 }
219
220 uint64_t now = Hardware::GetTime();
221 if (now - last_packet_time > 3000) {
222 ControlLogic::UpdateConnectivity(false);
223 }
224 }
