240 void MC_PULL_ONLINE_read() {
241 float *data = Hardware::ADC_GetValues();

243

245 MC_ONLINE_key_stu_raw[3] = data[1];

249 MC_ONLINE_key_stu_raw[1] = data[5];
250 MC_PULL_stu_raw[0] = data[6];
251 MC_ONLINE_key_stu_raw[0] = data[7];
252
253 for (int i = 0; i < 4; i++) {
254 if (MC_PULL_stu_raw[i] > PULL_voltage_up) MC_PULL_stu[i] = 1;
255 else if (MC_PULL_stu_raw[i] < PULL_voltage_down) MC_PULL_stu[i] = -1;
256 else MC_PULL_stu[i] = 0;
257
258 if (is_two == false) {
259 if (MC_ONLINE_key_stu_raw[i] > 1.65f) MC_ONLINE_key_stu[i] = 1;
260 else if (MC_ONLINE_key_stu_raw[i] < 1.55f) MC_ONLINE_key_stu[i] = 0;
261 } else {
262 if (MC_ONLINE_key_stu_raw[i] < 0.6f) MC_ONLINE_key_stu[i] = 0;
263 else if (MC_ONLINE_key_stu_raw[i] < 1.4f) MC_ONLINE_key_stu[i] = 3;
264 else if (MC_ONLINE_key_stu_raw[i] > 1.7f) MC_ONLINE_key_stu[i] = 1;
265 else MC_ONLINE_key_stu[i] = 2;
266 }
267
268 // Sync status
269 if (MC_ONLINE_key_stu[i] != 0) {
270 if (data_save.filament[i].status == AMS_filament_status::offline) {
271 data_save.filament[i].status = AMS_filament_status::online;
272 }
273 } else {
274 data_save.filament[i].status = AMS_filament_status::offline;
275 }
276
277 // Update pressure (voltage in mV)
278 data_save.filament[i].pressure = (uint16_t)(MC_PULL_stu_raw[i] * 1000.0f);

306 float Calculate(float E, float time_E) {
307 I_save += I * E * time_E;
308 if (I_save > pid_range) I_save = pid_range;
309 else if (I_save < -pid_range) I_save = -pid_range;
310
311 float output;
312 if (time_E != 0) output = P * E + I_save + D * (E - E_last) / time_E;
313 else output = P * E + I_save;
314
315 if (output > pid_MAX) output = pid_MAX;
316 if (output < pid_MIN) output = pid_MIN;
317 E_last = E;
318 return output;
319 }

366 float GetXByPressure(float pressure_voltage, float control_voltage, float time_E, pressure_control_enum control_type, float sign) {
367 float x=0;
368 switch (control_type) {
369 case pressure_control_enum::all:
370 x = sign * PID_pressure.Calculate(MC_PULL_stu_raw[CHx] - control_voltage, time_E);

382 x = x * x / 250;
383 } else {

398 void Run(float time_E) {
399 // Track distance for all movement types if needed, but critical for Klipper Move
400 // speed_as5600 is in mm/s? AS5600_Update says: speedx = dist_E / time_E. dist_E is mm.
401 // So speed_as5600 is mm/s.
402
403 float dist_step = fabs(speed_as5600[CHx] * time_E);
404 if (is_backing_out) {
405 last_total_distance[CHx] += dist_step;
406 }
407 if (motion == filament_motion_enum::velocity_control && target_distance > 0) {
408 accumulated_distance += dist_step;
409 if (accumulated_distance >= target_distance) {
410 SetMotion(filament_motion_enum::stop);
411 // TODO: Send async completion event? Or just stop.
412 // Klipper polls or waits?
413 // The spec says MOVE returns when done? Or "TELEMETRY".
414 // If open-loop, success is assumed.
415 // But effectively we are closed-loop on distance now.
416 }
417 }
418
419 float speed_set = 0;
420 float now_speed = speed_as5600[CHx];
421 float x = 0;
422
423 if (motion == filament_motion_enum::pressure_ctrl_idle) { // Idle
424 bool pid_invert = false;
425 switch(CHx) {
426 case 0: pid_invert = MOTOR_PID_INVERT_CH1; break;
427 case 1: pid_invert = MOTOR_PID_INVERT_CH2; break;
428 case 2: pid_invert = MOTOR_PID_INVERT_CH3; break;
429 case 3: pid_invert = MOTOR_PID_INVERT_CH4; break;
430 }
431 float pid_sign = dir * (pid_invert ? -1.0f : 1.0f);
432
433 if (MC_ONLINE_key_stu[CHx] == 0) {
434 Assist_send_filament[CHx] = true;
435 }
436 if (Assist_send_filament[CHx] && is_two) {
437 if (MC_ONLINE_key_stu[CHx] == 2) x = -dir * 666;
438 else if (MC_ONLINE_key_stu[CHx] == 1) {
439 // Timer logic...
440 }
441 } else {
442 if (MC_ONLINE_key_stu[CHx] != 0 && MC_PULL_stu[CHx] != 0) {
443 x = pid_sign * PID_pressure.Calculate(MC_PULL_stu_raw[CHx] - 1.65, time_E);
444 } else {
445 x = 0; PID_pressure.Clear();
446 }
447 }
448 } else if (MC_ONLINE_key_stu[CHx] != 0 || motion == filament_motion_enum::velocity_control) { // Allow vel control without sensor
449 bool pid_invert = false;
450 switch(CHx) {
451 case 0: pid_invert = MOTOR_PID_INVERT_CH1; break;
452 case 1: pid_invert = MOTOR_PID_INVERT_CH2; break;
453 case 2: pid_invert = MOTOR_PID_INVERT_CH3; break;
454 case 3: pid_invert = MOTOR_PID_INVERT_CH4; break;
455 }
456 float pid_sign = dir * (pid_invert ? -1.0f : 1.0f);
457
458 if (motion == filament_motion_enum::pressure_ctrl_in_use) {
459 if (pull_state_old) {
460 if (MC_PULL_stu_raw[CHx] < 1.55) pull_state_old = false;
461 } else {
462 if (MC_PULL_stu_raw[CHx] < 1.65) x = GetXByPressure(MC_PULL_stu_raw[CHx], 1.65, time_E, pressure_control_enum::less_pressure, pid_sign);
463 else if (MC_PULL_stu_raw[CHx] > 1.7) x = GetXByPressure(MC_PULL_stu_raw[CHx], 1.7, time_E, pressure_control_enum::over_pressure, pid_sign);
464 }
465 } else {
466 if (motion == filament_motion_enum::stop) {
467 PID_speed.Clear(); Hardware::PWM_Set(CHx, 0); return;
468 }
469 if (motion == filament_motion_enum::send) {
470 if (device_type_addr == BambuBus_AMS_lite) {
471 if (MC_PULL_stu_raw[CHx] < PULL_VOLTAGE_SEND_MAX) speed_set = MOTOR_SPEED_AMS_LITE_SEND; else speed_set = 0;
472 } else speed_set = MOTOR_SPEED_SEND;
473 }
474 if (motion == filament_motion_enum::slow_send) speed_set = MOTOR_SPEED_SLOW_SEND;
475 if (motion == filament_motion_enum::pull) speed_set = -MOTOR_SPEED_PULL;
476 if (motion == filament_motion_enum::velocity_control) speed_set = target_velocity; // Use Override
477
478 x = dir * PID_speed.Calculate(now_speed - speed_set, time_E);
479 }
480 } else {
481 x = 0;
482 }
483
484 if (x > 10) x += pwm_zero;
485 else if (x < -10) x -= pwm_zero;
486 else x = 0;
487
488 if (x > 1000) x = 1000;
489 if (x < -1000) x = -1000;
490
491 Hardware::PWM_Set(CHx, (int)x);
492 UpdateLEDStatus();
493 }

694 void AS5600_Update(float time_E) {
695 MC_AS5600.updata_angle();
696 for(int i=0; i<4; i++) {
697 if (!MC_AS5600.online[i]) {
698 speed_as5600[i] = 0; as5600_distance_save[i] = 0; continue;
699 }
700 int32_t now = MC_AS5600.raw_angle[i];
701 int32_t last = as5600_distance_save[i];
702 int cir_E = 0;
703 if (now > 3072 && last <= 1024) cir_E = -4096;
704 else if (now <= 1024 && last > 3072) cir_E = 4096;
705
706 float dist_E = -(float)(now - last + cir_E) * AS5600_PI * 7.5 / 4096;
707 as5600_distance_save[i] = now;
708
709 float speedx = dist_E / (time_E > 0 ? time_E : 0.001f);
710 speed_as5600[i] = speedx;
711
712 data_save.filament[i].meters += dist_E / 1000.0f;
713 }

1089 motors[num].SetMotion(filament_motion_enum::pull);
1090 }
1091 break;

1100 {
1101 is_backing_out = false;
1102 pull_state_old = true;
1103 filament_now_position[num] = filament_using;
1104 time_end = time_now + 1500;
1105 }
1106 else if (filament_now_position[num] == filament_using)
1107 {

1111 Hardware::LED_SetColor(num, 0, 255, 255, 255);
