41bool BambuBusProtocol::ParseByte(uint8_t byte) {
42 static int _index = 0;
43 static int length = 999;
44 static uint8_t data_length_index;
45 static uint8_t data_CRC8_index;
46
47 packet_ready = false;
48
49 if (_index == 0) // waiting for first data
50 {
51 if (byte == 0x3D) // 0x3D-start
52 {
53 rx_buf[0] = 0x3D;
54 _RX_IRQ_crcx.restart();
55 _RX_IRQ_crcx.add(0x3D);
56 data_length_index = 4;
57 length = data_CRC8_index = 6;
58 _index = 1;
59 }
60 return false;
61 }
62 else // have 0x3D, normal data
63 {
64 rx_buf[_index] = byte;
65 if (_index == 1) // package type byte
66 {
67 if (byte & 0x80) // short head package
68 {
69 data_length_index = 2;
70 data_CRC8_index = 3;
71 }
72 else // long head package
73 {
74 data_length_index = 4;
75 data_CRC8_index = 6;
76 }
77 }
78 if (_index == data_length_index) // the length byte
79 {
80 length = byte;
81 }
82 if (_index < data_CRC8_index) // before CRC8 byte, add data
83 {
84 _RX_IRQ_crcx.add(byte);
85 }
86 else if (_index == data_CRC8_index) // the CRC8 byte, check
87 {
88 if (byte != _RX_IRQ_crcx.calc()) // check error, return to waiting 0x3D
89 {
90 _index = 0;
91 return false;
92 }
93 }
94 ++_index;
95 if (_index >= length) // recv over
96 {
97 _index = 0;
98 rx_len = length;
99 packet_ready = true;
100 return true;
101 }
102 if (_index >= 999) // buffer overflow protection
103 {
104 _index = 0;
105 }
106 }
107 return false;
108}

142bool BambuBusProtocol::CheckCRC16(uint8_t *data, int data_length)
143{
144 crc_16.restart();
145 data_length -= 2;
146 for (auto i = 0; i < data_length; i++)
147 {
148 crc_16.add(data[i]);
149 }
150 uint16_t num = crc_16.calc();
151 if ((data[(data_length)] == (num & 0xFF)) && (data[(data_length + 1)] == ((num >> 8) & 0xFF)))
152 return true;
153 return false;
154}

167BambuBus_package_type BambuBusProtocol::IdentifyPacket(uint8_t* buf, uint16_t length) {
168 if (CheckCRC16(buf, length) == false)
169 {
170 return BambuBus_package_type::NONE;
171 }
172 if (buf[1] == 0xC5)
173 {
174 switch (buf[4])
175 {
176 case 0x03: return BambuBus_package_type::filament_motion_short;
177 case 0x04: return BambuBus_package_type::filament_motion_long;
178 case 0x05: return BambuBus_package_type::online_detect;
179 case 0x06: return BambuBus_package_type::REQx6;
180 case 0x07: return BambuBus_package_type::NFC_detect;
181 case 0x08: return BambuBus_package_type::set_filament_info;
182 case 0x20: return BambuBus_package_type::heartbeat;
183 default: return BambuBus_package_type::ETC;
184 }
185 }
186 else if (buf[1] == 0x05)
187 {
188 long_packge_data info;
189 ParseLongPacket(buf, length, &info);
190
191 switch (info.type)
192 {
193 case 0x21A: return BambuBus_package_type::MC_online;
194 case 0x211: return BambuBus_package_type::read_filament_info;
195 case 0x218: return BambuBus_package_type::set_filament_info_type2;
196 case 0x103: return BambuBus_package_type::version;
197 case 0x402: return BambuBus_package_type::serial_number;
198 default: return BambuBus_package_type::ETC;
199 }
200 }
201 return BambuBus_package_type::NONE;
202}

231void BambuBusProtocol::BuildPacketWithCRC(uint8_t *data, uint16_t &data_length)
232{
233 crc_8.restart();
234 if (data[1] & 0x80)
235 {
236 for (auto i = 0; i < 3; i++) crc_8.add(data[i]);
237 data[3] = crc_8.calc();
238 }
239 else
240 {
241 for (auto i = 0; i < 6; i++) crc_8.add(data[i]);
242 data[6] = crc_8.calc();
243 }
244 crc_16.restart();
245 // length calculation logic for CRC16
246 // In original code, data_length was passed IN as the target length including CRC16 bytes placeholders?
247 // "data_length += 2;" at the end of original function suggests the input length EXCLUDED the CRC16.
248 // "data_length -= 2;" in loop suggests it iterated up to CRC pos.
249
250 // Let's assume input data_length is the length of payload + headers, WITHOUT CRC16.
251 // Wait, original code:
252 // package_send_with_crc(uint8_t *data, int data_length)
253 // data_length -= 2; ... loop i < data_length ...
254 // data[data_length] = ...
255 // data_length += 2;
256 // So the input data_length INCLUDED the 2 bytes valid space for CRC16.
257
258 int loop_len = data_length - 2;
259 for (auto i = 0; i < loop_len; i++)
260 {
261 crc_16.add(data[i]);
262 }
263 uint16_t num = crc_16.calc();
264 data[loop_len] = num & 0xFF;
265 data[loop_len + 1] = num >> 8;
266 // Length is already correct (it was passed in including the 2 bytes space)
267}

280void BambuBusProtocol::BuildLongPacket(long_packge_data *data, uint8_t *out_buffer, uint16_t &out_length)
281{
282 out_buffer[0] = 0x3D;
283 out_buffer[1] = 0x05;
284 data->package_length = data->data_length + 15; // 13 header + 2 crc16 ?
285 // Header is: 3D 00 (2) + struct(11) = 13?
286 // struct is 11 bytes: num(2)+len(2)+crc8(1)+target(2)+source(2)+type(2) = 11.
287 // total header = 3D 00 + 11 = 13.
288 // + data
289 // + crc16 (2)
290 // = 15 + data_length. Correct.
291
292 memcpy(out_buffer + 2, data, 11);
293 memcpy(out_buffer + 13, data->datas, data->data_length);
294 out_length = data->data_length + 15;
295 BuildPacketWithCRC(out_buffer, out_length);
296}
